    function hub(pattern) {
    var callsite = getCallSite();
    var recursive = inhub;
    inhub = true;

    loadFiles(pattern, path.dirname(callsite));

    // only create the task tree once
    if (!recursive) {
        for (var name in tasks)
            addTask(tasks[name]);

        inhub = false;
    }
}

  • description and source-code
    isValidHubFile = function (file) {
    if (
        _.isPlainObject( file ) &&
        file.relativePath && typeof file.relativePath === 'string' &&
        file.absolutePath && typeof file.absolutePath === 'string' &&
        file.uniqueName   && typeof file.uniqueName   === 'string'
    ){
        return true;
    }
    return false;
}
